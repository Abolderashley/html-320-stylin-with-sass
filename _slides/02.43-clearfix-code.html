--- layout: slide title: "Clearfix SCSS Code" --- 
    
@mixin clearfix {
  &:before, &:after {
    content: " ";
    display: table;
  }
  &:after {
    clear: both;
  }
}
.row {
  @include clearfix;
  margin-left: 10px;
  margin-right: 10px;
}
.row > .box {
  float: left;
  padding: 10px;
}
