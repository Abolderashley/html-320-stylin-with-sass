<div class="left-align">
  <div class="">
    <p>SCSS:</p>
<pre><code>$background: #f3f3f3;
$text_color: darken($background, 30%);
$accent_color: lighten($text_color, 10);
body {
  color: $text_color;
  background: $background;
  a {
    color: $accent_color;
  }
}</code></pre>
  </div>
  <div>
    <p>CSS:</p>
<pre><code>body {
  color: #a7a7a7;
  background: #f3f3f3; }
body a {
  color: silver; }</code></pre>
  </div>
</div>
